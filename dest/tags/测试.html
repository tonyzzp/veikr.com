t.cs

  1: using System;
  2: 
  3: public class T{
  4:     public static void Main(){

  7:         TimeSpan   ts1   =   DateTime.Now-   dt1;
  8:         long start=(long)ts1.TotalMilliseconds;
  9:         for(long i=0;i<=100000000;i++){
 10:             a+=i;
 11:         }

 19: }
